ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'password';
flush privileges;

DROP TABLE IF EXISTS `notification`;

CREATE TABLE `notification` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT,
  `target` varchar(255) NOT NULL,
  `read` bool DEFAULT FALSE,
  `event_name` varchar(255) NOT NULL,
  `actor_addr` varchar(255) NOT NULL,
  `actor_name` varchar(255) NULL,
  `actor_avatar` varchar(255) NULL,
  `text` varchar(255) NULL,
  `image` varchar(255) NULL,
  `timestamp` bigint NOT NULL,
  `item_id` int NULL,
  `item_type` int NULL,
  `item_status` int NULL,
  `item_flag` bool DEFAULT FALSE,
  `item_sender` varchar(255) NULL,
  `item_receiver` varchar(255) NULL,
  `item_data` varchar(1024) NULL,
  `extra_data` varchar(1024) NULL,
  PRIMARY KEY (`id`)
);
